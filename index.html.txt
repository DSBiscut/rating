    </div> <script>
        const userNameInput = document.getElementById('userName');
        const movieNameInput = document.getElementById('movieName');
        const feedbackMessage = document.getElementById('feedback-message');
        const categories = ['visuals', 'sound', 'comfort', 'story', 'overall'];

        // Function to load saved data from LocalStorage
        window.onload = () => {
            const savedReview = JSON.parse(localStorage.getItem('pvrReview'));
            if (savedReview) {
                userNameInput.value = savedReview.userName || '';
                movieNameInput.value = savedReview.movieName || '';

                categories.forEach(category => {
                    if (savedReview.ratings && savedReview.ratings[category]) {
                        const rating = savedReview.ratings[category];
                        const radio = document.querySelector(`input[name="${category}"][value="${rating}"]`);
                        if (radio) {
                            radio.checked = true;
                        }
                    }
                });
                updateFeedbackMessage(savedReview.userName);
            }
        };

        function submitReview() {
            const userName = userNameInput.value.trim();
            const movieName = movieNameInput.value.trim();
            const ratings = {};
            let allRatingsGiven = true;

            if (!userName || !movieName) {
                feedbackMessage.textContent = "Please enter your Name and Movie Name.";
                feedbackMessage.style.color = var(--pvr-red);
                return;
            }

            categories.forEach(category => {
                const selectedRating = document.querySelector(`input[name="${category}"]:checked`);
                if (selectedRating) {
                    ratings[category] = selectedRating.value;
                } else {
                    allRatingsGiven = false;
                }
            });

            if (!allRatingsGiven) {
                feedbackMessage.textContent = "Please give a rating for all categories.";
                feedbackMessage.style.color = var(--pvr-red);
                return;
            }

            const reviewData = {
                userName: userName,
                movieName: movieName,
                ratings: ratings,
                timestamp: new Date().toISOString()
            };

            localStorage.setItem('pvrReview', JSON.stringify(reviewData));
            updateFeedbackMessage(userName);
        }

        function clearRatings() {
            userNameInput.value = '';
            movieNameInput.value = '';
            categories.forEach(category => {
                document.querySelectorAll(`input[name="${category}"]`).forEach(radio => radio.checked = false);
            });
            localStorage.removeItem('pvrReview');
            feedbackMessage.textContent = '';
        }

        function updateFeedbackMessage(name) {
            feedbackMessage.textContent = `Thank you for your feedback, ${name || 'User'}!`;
            feedbackMessage.style.color = var(--gold);
        }

    </script>
